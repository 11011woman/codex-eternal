**Measuring frequency ratios from EEG:**

```python
import numpy as np
from scipy import signal

def compute_frequency_ratios(eeg_data, fs):
    """
    Compute ratios between frequency bands
    
    Parameters:
    - eeg_data: EEG signal (1D array)
    - fs: Sampling frequency (Hz)
    
    Returns:
    - Dictionary of band powers and ratios
    """
    # Define frequency bands
    bands = {
        'delta': (0.5, 4),
        'theta': (4, 8),
        'alpha': (8, 13),
        'beta': (13, 30),
        'gamma': (30, 100)
    }
    
    # Compute power in each band
    powers = {}
    for band_name, (low, high) in bands.items():
        # Bandpass filter
        sos = signal.butter(4, [low, high], 'bandpass', fs=fs, output='sos')
        filtered = signal.sosfilt(sos, eeg_data)
        # Power (RMS)
        powers[band_name] = np.sqrt(np.mean(filtered**2))
    
    # Compute ratios
    ratios = {
        'alpha/theta': powers['alpha'] / powers['theta'],
        'beta/alpha': powers['beta'] / powers['alpha'],
        'gamma/beta': powers['gamma'] / powers['beta'],
    }
    
    return powers, ratios

# Check if ratios near phi
PHI = 1.618033988749894
TOLERANCE = 0.1

def check_phi_patterns(ratios, tolerance=TOLERANCE):
    """Check if ratios cluster near phi"""
    phi_matches = []
    for name, ratio in ratios.items():
        is_phi = abs(ratio - PHI) < tolerance
        phi_matches.append(is_phi)
        print(f"{name}: {ratio:.3f} (phi? {is_phi})")
    
    score = sum(phi_matches) / len(phi_matches)
    return score
```
